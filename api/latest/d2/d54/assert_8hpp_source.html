67  ((ALLOY_LIKELY(condition)) \
68  ? ((void)0) \
69  : []( auto file_name, \
70  auto line_number, \
71  auto function_name ) \

162  // Note: This function is defined inline intentionally.
163  //
164  // For GCC/Clang, ALLOY_BREAKPOINT does not trigger the debugger if this
165  // definition is compiled into a different shared library than the consuming
166  // binary is. When inline, the breakpoint is visible, and brings up the
167  // debugger as desired.
168  //
169  // This hasn't been tested with MSVC; but the behavior has been normalized
170  // here since it's required in order to use this with two of the target
171  // compilers.
172  [[noreturn]]
173  inline void assert_internal( const char* message,
174  const char* file_name,
175  std::size_t line,
176  const char* function_name )
177  {
178  ::std::fprintf(stderr, "[assertion] %s (%zu)::%s\n"
179  " %s\n",
180  file_name,
181  line,
182  function_name,
183  message );
184  std::fflush(stderr);
185  // TODO(bitwizeshift): print out stack-trace on assertion failure
186 
187  ALLOY_BREAKPOINT();
188  ::std::terminate();
189  }
