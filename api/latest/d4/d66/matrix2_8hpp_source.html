520 inline constexpr alloy::core::matrix2
521  alloy::core::matrix2::inverse()
522  const noexcept
523 {
524  const auto det = determinant();
525 
526  if (det == real{0}) {
527  return matrix2{};
528  }
529 
530  const auto inv_det = (real{1} / det);
531  return matrix2{
532  get(1,1) * inv_det, -get(1,0) * inv_det,
533  -get(0,1) * inv_det, get(0,0) * inv_det
534  };
535 }
