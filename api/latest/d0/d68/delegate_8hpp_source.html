265  template <typename T>
266  using delegate_forward_result_t = std::conditional_t<
267  std::is_reference<T>::value, T, std::add_rvalue_reference_t<T>
268  >;
269 
270  // A small utility for forwarding class template arguments in a variadic
271  // member function.
272  //
273  // This allow by-value move-only types to be propagates to the underlying
274  // function (e.g. for if std::unique_ptr is a parameter).
275  //
276  // * l-value references are passed-through as l-value references
277  // * r-value references are passed-through as r-value references
278  // * By-value arguments are std::moved to r-value references
279  template <typename T, typename U>
280  constexpr delegate_forward_result_t<T> delegate_forward(U&& u)
281  {
282  return static_cast<delegate_forward_result_t<T>>(u);
283  }
284 
285 } // namespace alloy::core::detail

291 template <typename R, typename...Args>
292 template <auto Fn, typename>
293 inline constexpr alloy::core::delegate<R(Args...)>
294  alloy::core::delegate<R(Args...)>::bind()
295  noexcept
296 {
297  const auto ptr = static_cast<const void*>(nullptr);
298  const auto fn = [](const void* p, Args...args) -> R
299  {
300  (void) p;
301 
302  if constexpr (std::is_void<R>::value) {
303  std::invoke(Fn, detail::delegate_forward<Args>(args)...);
304  } else {
305  return std::invoke(Fn, detail::delegate_forward<Args>(args)...);
306  }
307  };
308 
309  return delegate{ptr, static_cast<function_type>(fn)};
310 }
