705 inline alloy::core::radian
706  alloy::core::vector2::angle_between( const vector2& other )
707  const noexcept
708 {
709  auto mag_product = magnitude() * other.magnitude();
710 
711  if( almost_equal( mag_product, real{0} ) ){
712  mag_product = default_tolerance;
713  }
714 
715  auto f = dot(other) / mag_product;
716 
717  f = clamp( f, real{-1.0}, real{1.0} );
718  return trigonometry::arccos( f );
719 }

746  alloy::core::vector2::normalize()
747  noexcept
748 {
749  const auto square_mag = square_magnitude();

751  if (almost_equal(square_mag, real{1})) {
