529 inline void
530  alloy::core::observable::remove(not_null<detail::observer_ptr_base*> p)
531  const noexcept
532 {
533  // If 'p' is the head, change the head pointer
534  if (p.get() == m_head) {
535  m_head = m_head->m_next;
536  return;
537  }
538 
539  // If 'p' is not the head, iterate to find the current entry and rebind
540  auto* prev = m_head;
541  auto* current = m_head->m_next;
542 
543  while (current != nullptr) {
544  if (current == p.get()) {
545  prev->m_next = current->m_next;
546  return;
547  }
548  prev = current;
549  current = current->m_next;
550  }
551 }

554 inline void
555  alloy::core::observable::rebind(not_null<detail::observer_ptr_base*> old_p,
556  not_null<detail::observer_ptr_base*> new_p)
557  const noexcept
558 {
559  // If 'old_p' is the head, rebind to new_p
560  if (old_p.get() == m_head) {
561  new_p->m_next = m_head->m_next;
562  m_head = new_p.get();
563  old_p->m_next = nullptr;
564  return;
565  }
566 
567  // If 'old_p' is not the head, iterate to find the current entry and rebind
568  auto* prev = m_head;
569  auto* current = m_head->m_next;
570 
571  while (current != nullptr) {
572  if (current == old_p.get()) {
573  prev->m_next = new_p.get();
574  new_p->m_next = current->m_next;
575  old_p->m_next = nullptr;
576  return;
577  }
578  prev = current;
579  current = current->m_next;
580  }
581 }
