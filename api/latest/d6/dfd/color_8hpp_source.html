591 inline constexpr alloy::core::color
592  alloy::core::blend( const color& c1, const color& c2 )
593 {
594  const auto diff = real{1} - c2.a();
595 
596  auto result = color{};
597  result.a() = c1.a() * diff + c2.a();
598  if (almost_equal(result.a(),0)) {
599  return color{0,0,0,0};
600  }
601  result.r() = (c1.r() * c1.a() * diff + c2.r() * c2.a()) / result.a();
602  result.g() = (c1.g() * c1.a() * diff + c2.g() * c2.a()) / result.a();
603  result.b() = (c1.b() * c1.a() * diff + c2.b() * c2.a()) / result.a();
604 
605  return result;
606 }
